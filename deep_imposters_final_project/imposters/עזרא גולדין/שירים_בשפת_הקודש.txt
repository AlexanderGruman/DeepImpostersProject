לגיסי המשכיל המהולל איש טהר לב ורוח נכון, סופר מהיר בשפת קדש ואוהב ספרותה, ה' מרדכי אסטרינסקי נ"י. ולאחי הצעיר ממני לימים, המופלג המפורסם, חריף ושנון, ושוקד על דברי תורה יומם ולילה, ה' ישראל גאלדין נ"י.


ולרעַי אוהבי הקרובים והרחוקים!

על מזבח אהבתכם אקריב היום את מנחתי הדלה, מנחת עני, קומץ עלים זה הנני מגיש לפניכם בלב מלא אהבה נאמנה, כי לא להעשיר ספרות העברית בשירים חדשים מגמת פני, כי יודע אני מך ערכי ומיעוט שנותי, אך למענכם רעי ואוהבי נגשתי אל הדפוס בשנת העשרים לימי חיי, הא לכם שירי המעטים אשר בקשתם, ואם אחד מכם ימצא בם טעם לשבח, אומר: די לי! ואוציא לאור את יתר שירי אשר אתי בכתובים. השתחויתי! אמצא נא חן בעיניכם קוראים יקרים, ומבקרים ישרים!…

אֲיֻמָּה כָּל שָׁעָה כָּל רֶגַע לָהּ נוֹרָא…

דְּרָשׁ לָהּ מַחֲסֶה מִצַּר וּמִשּׁוֹדֵד מָנוֹחַ..!

(ספור היסתורי)


1.
וַיְהִי שֵׁם רֶעכֶען הָעִבְרִי לִתְהִלָּה בָּאָרֶץ,

אַף כִּי הִשָּׁהוּ אֵל חָכְמָה וְתוֹרָתוֹ לֹא רַבָּה,

אַךְ הַשְׂכָּלָתוֹ מִקֶּרֶב לִבּוֹ פָּרְצָה פָּרֶץ,

לֹא מִשִּׁמְשָׁהּ לוֹ קַרְנַיִם אַךְ קִרְבּוֹ הַלַּבָּה.


2.
לִכְבוֹדוֹ שָׁר לֶעסְסִינְג, מִיצְקֶעוִיץ אֶת שִׁירוֹ,

“נָתָן הֶחָכָם” “הוֹד טַאדֶעאוּשׁ” לִכְבוֹדוֹ נִבְרָאוּ

כִּי שֶׁמֶשׁ חִנּוֹ זָרַח לֹא רַק בִּבְנֵי עִירוֹ,

לְקַרְנֵי הוֹדוֹ מְשׁוֹרְרֵי אֶרֶץ תָּמָהוּ…


3.
וּכְגֲי הַשִּׁלֹּחַ דּוּמָם יִתְגַּלְגָּלוּ,

עַל חֲלֻקֵּי הַנַּחַל וַעֲרוּצֵי הַנְּחָלִים,

כֵּן חַיֵּי רֶעכֶען שַׁלְוַת עוֹלָם שָׁלוּ,

לֹא עָדָה עֲלֵיהֶם אָסוֹן לֹא שַׁוְעַת אֻמְלָלִים.


4.
וַתְּהִי לוֹ אִשָּׁה אֲהוּבָה בִּינָה הַיָּפָה,

עֵינֶיהָ כּוֹכָבִים אַף מִצְחָהּ יָרֵחַ,

אַפָּהּ מַעֲשֵׂה אֳמָן חוּט שָׁנִי הַשָּׂפָה,

כֻּלָּהּ עֲלִיזָה כִּצְחוֹק, כָּאוֹר שָׂמֵח…


5.
וַיְיָ סָגַר אֶת רַחְמָהּ וַתְּהִי בִּינָה עֲקָרָה,

אַךְ כָּל לַהַב לֵב אִשָּׁה לְרֶעכֶען שָׁמָרָה,

לֹא חִלְלָה יָפְיָהּ וְתֻמָּתָהּ אַךְ בָּרָה,

גַּם הוּא כְּבָבַת עַיִן יוֹם וַָלַיִל נְצָרָהּ!


6.
קֶמֶט דַּק עַל מִצְחָהּ אֶת לִבּוֹ הִרְתִּיחַ,

צֵל קַל עַל לֶחֱיָהּ חִישׁ עֵינָיו חָשָׁכוּ,

מַבָּט אֶחָד נָעִים תֵּבֵל כֻּלָּהּ הִשְׁכִּיחַ,

וְעָבֵי תוּגָה כִּצְלָלִים בָּרָחוּ…


7.
וְדוּמָם כִּי יָשְׁבוּ אֶל חַלּוֹן הַפָּתוּחַ,

צַהֲלַת יְלָדִים בָּרְחוֹב אַךְ אָזְנָם קָשָׁבָה,

אָז מִקֶּרֶב לִבּוֹת שְׁנֵיהֶם יָצְאָה הָרוּחַ,

דִּמְעָתָהּ בֶּעָיִן וְאוֹר עֵינוֹ כָּבָה…


8.
הַשֶּׁמֶשׁ נָטָה לַעֲרוֹב הַצְּלָלִים נִטָּיוּ,

מַלְאֲכֵי שָׁלוֹם יְרַחֲפוּ עַל פְּנֵי תֵבֵל כֻּלָּהּ,

בֵּין עֳפָאֵי הָעֵצִים כָּל צִפּוֹר יֶהֱמָיוּ,

בָּאָחוּ וַאֲגַמִּים אַךְ תְּרוּעָה וַהֲמֻלָּה…


9.
לִשְׁנֵי סוּסִים אַבִּירִים רְתוּמָה מֶרְכָּבָה,

בְּלֵב יַעַר אָפֵל תְּשָׂרֵךְ דַּרְכָּהּ בַּעֲצַלְתַּיִם,

בָּהּ רֶעכֶען עִם רַכָּבוֹ טָבוּעַ בַּמַּחֲשָׁבָה:

“מִי יָקוּם אַחֲרַי כִּי אֶפָּרֵד מִן הַחַיּיִם?..!”


10.
נוֹרָא זֶה הָרַעֲיוֹן גַּם לֶעָנִי וָחֵלֶךְ,

כִּי יִזְכּוֹר מְרוּדוֹ בַּתֵּבֵל בֵּית סֻבֳּלוֹ,

וּבְבוֹא יוֹמוֹ הַמַּר מִתֵּבֵל כִּי יֵלֶךְ,

אֵין זוֹכֵר אֶת עֵץ הַיָּבֵשׁ, מַר גּוֹרָלוֹ!


11.
אַךְ אָיוֹם זֶה הָרַעְיוֹן עֲשֶׂרֶת מוֹנִים,

לְדִשְׁנֵי אֶרֶץ עַל כָּל טוּב יִתְעַנָּגוּ…

אֶל מוֹתוֹ יְיַחֵלוּ קְרוֹבִים אֶבְיוֹנִים,

“מָתַי יָבוֹא יוֹמוֹ?” יוֹם וָלַיִל יִדְאָגוּ…


12.
וּבְעוֹד מֵתָם כְּאֶבֶן מֻטָּל לִפְנֵיהֶם,

עוֹד לוֹ שִׁבֳּלֵי תֶבֶן כְּמַחַט בִּבְשַׂר הַחַי,

לִבָּם בָּם יִחַד אֶל בִּצְעוֹ עֵינֵיהֶם,

וּשְׂמֵחִים יְלַוּוּהוּ לַקֶּבֶר לַגַּיְא…


13.
וְיֹאבַד שֵׁם וָזֵכֶר לוֹ פֹּה בָּאָרֶץ,

מִתְעַנְּגִים עַל טוּבוֹ לֹא נַפְשָׁם כֹּאָבֶת,

אַךְ זֶה מוֹת־עוֹלָם! נוֹרָא הַקָּרֶץ,

כִּי זֵכֶר לַגֶּבֶר אַךְ עַל קֶבֶר מַצָּבֶת!..


14.
כֹּה יָשַׁב רֶעכֶען תָּפוּשׂ בִּשְׂעִפֵּי עֶצֶב,

רַעְיוֹנָיו כַּיֶּלֶק פָּשְׁטוּ וַיָּעוּפוּ:

"מִי יָקוּם אַחֲרָיו כִּי יָבוֹא לוֹ קֶצֶב,

כִּי יֶחְשְׁכוּ הָרֹאוֹת, לִבּוֹ, רוּחוֹ יָסוּפוּ?.."


15.
וּכְנֶשֶׁר נִבְעָת פִּתְאֹם מֵרַעַם הַמּוֹרֶה,

כַּאֲרִי נִרְדָּם יֵעוֹר מֵחֵץ מָוֶת בּוֹ פָגַע,

כֵּן נֵעוֹר חִישׁ רֶעכֶען בָּרַעַד וּמוֹרָא,

“רַכָּב עֲמוֹד – קָרָא – עֲצָר נָא כְּרָגַע”.


16.
“קוֹל עוֹלָל בֹּכֶה מַר שָׁמֹעַ שָׁמַעְתִּי”,

“הָבָה אֲחַפֵּשׂ וְאֶרְאֶנּוּ בָּעַיִן”,

אֵל שָׁלַח לִי נֹחַם – הָגָה לִבּוֹ – יָדַעְתִּי,

לְהַמְתִּיק מְרִירוּת עָצְבִּי בַּחַיִּין…


17.
וַיְקַפֵּץ כַּאֲרִי אֶל לֵב הַיָּעַר,

לְבַקֵּשׁ אֶת הָעוֹלָל וַיִּמְצָאֶנּוּ…

רַעַד בָּא אֶל לֵב הָרַכָּב הַבָּעַר,

לְפִי הַהַגָּדָה – אָמַר – יְלוּד אִישׁ אֵינֶנּוּ…


18.
וַיַּרְא רֶעכֶען וְהִנֵּה יַלְדָּה נְעִימָה,

מַרְאֶהָ כַּשֶּׁמֶשׁ כַּחַמָּה אֲהוּבָה…

שְׂפָתֶיהָ שׁוֹשַׁנִּים וְעַל עָיִן דִּמְעָה…

עַל צַוָּאר נֹפֶךְ זָהָב… וּמוֹדָעָה כְּתוּבָה…


19.
בְּלִי רַעַשׁ אוֹפָן וָרֶכֶב הָעֵת עוֹבָרֶת,

בְּלִי כֹחַ קִיטוֹר מֶרְכַּבְתָּהּ עָפָה,

הַמֶּרְכָּבָה הִיא הַתֵּבֵל דּוּמָם נִגְרֶרֶת,

בְּכֹחַ פֶּלִאי בְּלִי מַעֲצוֹר נֶהֱדָפָה…


20.
בֶּן שָׁלֹש הַיֶּלֶד עַל עַפְרָהּ יִשְׁתַּעֲשֵעַ,

וּבְעֵינָיו הַטְּהוֹרוֹת יַבִּיט לָרָקִיעַ,

עוֹד יֶחֱזֶה שָׁמַיִם עֵת יִגְוַע גָּוֹעַ

בֶּן שְׁמוֹנִים הַזָּקֵן כִּי שְׁאוֹל יַצִּיעַ.


21.
וְיֵרֶא, הֵמָּה לֹא כֶעָשָׁן נִמְלָחוּ,

וָאָרֶץ גַּם הִיא עַל עָמְדָהּ עוֹמֶדֶת.

רַק יְמֵי אֱנוֹשׁ כַּמַּיִם יִבְרָחוּ,

רַק חַיֵּי בֶּן־תְּמוּתָה רוּחַ נֹדֶדֶת…


22.
“בִּינָה” הָעֲקָרָה כְּבָר אֵינָהּ בַּחַיִּים,

עַל פְּנֵי רֶעכֶען נִקְבְּרָה שִׁמְשׁוֹ הַטְּהוֹרָה,

אַךְ מַלְאַךְ־נִחֻמִּים שָׁלְחוּ לוֹ שָׁמַיִם,

לְתָמְכוֹ לְעֵת זִקְנָה זוּ הָעַלְמָה צִפּוֹרָה.


23.
וַיֶּאֱהָבֶהָ כְּנַפְשׁוֹ תַּחַת בַּת וָאֵשֶׁת,

דַּעַת וְלָשׁוֹן לִמְּדָהּ וַתְּהִי מַשְׂכֶּלֶת,

מִדְבָּרָהּ חֵץ אַהֲבָה וְעֵינָהּ הַקֶּשֶׁת…

כֻּלָּהּ כָּזַיִת רַעֲנָן מְלֵאָה תוֹחֶלֶת.


24.
בַּחוּרֵי יֹפִי רָאוּהָ כֵּן תָּמָהוּ,

בְּלִבָּם מוֹקֵד עוֹלָם רֹאשָׁם כָּפָפוּ,

נִשְׁמַת חַיֵּיהֶם בְּאַהֲבָה לָהּ נָשָׂאוּ,

וּבִכְאֵב נֶצַח שָׁבוּ אֲלֵיהֶם אָסָפוּ…


25.
גַּם אַדִּיר הַכֶּסֶף וּמִשְׂגַּב הַדַּעַת,

ר….. הֶחָכָם הַנּוֹדָע לִתְהִלָּה,

לְרֶעכֶען כִּי שָׁלַח לִכְרֹה אֹזֶן שׁומַעַת,

“זֹאת לֹא תֶהִי” – הָיְתָה הַמִּלָּה…


26.
יַד אַהֲבָה בַּרְזֶל, חֹק אֵל מִצְוָתָהּ,

בְּלִבּוֹת הַנְּעוּרִים הַדָּם הִיא שָׂרָפָה,

זֶה יָמִים וִירָחִים אַהֲבָה הָגָתָה,

לְמֶעדְרִינָאוְוסְקִי הַפּוֹלַנִּי צִפּוֹרָה הַיָּפָה.


27.
אַהֲבָה בְּלִי תִקְוָה – בְִּלִי שֶׂכֶר יְסוּרִים,

תֶּמַח אוֹר הַתֹּאַר צְחוֹק שָׂפָה שׂרֶפֶת,

עַל פְּנֵי עַלְמָה בָּרָה תַּעַשׂ לָהּ תַּמְרוּרִים,

וְנֶפֶשׁ דֹּד עֶלֶם כָּאַגְמוֹן כּוֹפֶפֶת…


28.
הַלַּיְלָה הִשְׁלִיךְ הָס בָּאָרֶץ וְרָקִיעַ,

הַכּוֹכָבִים מִמְּסִלוֹתָם כָּעֵינַיִם יַבִּיטוּ…

וּמֶעדְרִינָאוְוסְקִי הֶאָדוֹן לִיפָתוֹ הִגִּיעַ,

בִּלְבָבוֹת דּוֹפְקִים יָד זֶה לָזֹה יוֹשִׁיטוּ.


29.
“אַתְּ לִי אוֹר נֹגַהּ” הִבִּיעַ בְּאֵשׁ בְּחֹם,

"זוּלָתֵךְ חַיַּי תֹּהוּ אַךְ הֶבֶל הַיָּמִים,

"תְּמוּנָתֵךְ אֶשְׂבְּעָה בְּהָקִיץ וּבַחֲלוֹם,

“הָהּ אֳהָבֵךְ צִפּוֹרָה אַהֲבַת עוֹלָמִים”.


30.
